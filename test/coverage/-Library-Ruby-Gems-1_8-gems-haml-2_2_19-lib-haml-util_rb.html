263     # This is used for methods in {Haml::Buffer} that need to be very fast,
 
264     # and take a lot of boolean parameters
 
265     # that are known at compile-time.
 
266     # Instead of passing the parameters in normally,
 
267     # a separate method is defined for every possible combination of those parameters;
 
268     # these are then called using \{#static\_method\_name}.
 
269     #

271     # All conditionals based on the static parameters
 
272     # are done as embedded Ruby within this template.
 
273     # For example:
 
274     #

276     #       <% if baz && bang %>

279     #         return foo - bar
 
280     #       <% else %>

282     #       <% end %>

286     #
 
287     # @overload def_static_method(klass, name, args, *vars, erb)
 
288     # @param klass [Module] The class on which to define the static method
 
289     # @param name [#to_s] The (base) name of the static method
 
290     # @param args [Array<Symbol>] The names of the arguments to the defined methods

293     #   to be made available to the ERB template

295     def def_static_method(klass, name, args, *vars)
 
296       erb = vars.pop
 
297       powerset(vars).each do |set|

304       end
